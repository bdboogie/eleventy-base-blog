                var tipPercent = 17;

                document.getElementById('tipText').innerHTML = 
                `Your total bill, including tip comes to £${(totalAmount + tipAmount).toFixed(2)}` +
                `. The tip amount was £${tipAmount.toFixed(2)}`;
